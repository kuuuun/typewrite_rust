5.1 match和if let

5.1.1 匹配模式: match和if let

enum Direction {
    East,
    West,
    North,
    South,
}

// 模式匹配最常用的就是 match 和 if let
fn main() {
    let dire = Direction::South;
    match dire {
        Direction::East => println!("East"),
        Direction::North | Direction::South => {
            println!("South or North");
        },
        _ => println!("West"),
    };
}

5.1.2 使用 match 表达式赋值

enum IpAddr {
   Ipv4,

}

    let ip1 = IpAddr::Ipv6;

#[derive(Debug)]

    match coin {
        Coin::Penny => 1,

            println!("State quarter from {:?}!", state);
            25
        },
    }
}

fn main() {
    let coin = Coin::Quarter(UsState::Alaska);
    assert_eq!(value_in_cents(coin), 25);
}

#[allow(dead_code)]
enum Action {
    Say(String),
    MoveTo(i32, i32),
    ChangeColorRGB(u16, u16, u16),
}

fn main() {
    let actions = Action::MoveTo(12, 34);
    let res = match actions {
        // 获取远祖，并拆包
        Action::MoveTo(x, y) => (Some(x), Some(y)),
        _ => (None, None),
    };
    assert_eq!(res.0.unwrap(),12);
    assert_eq!(res.1.unwrap(),34);
}

5.2 解构Option

#![allow(unused)]
fn main() {
// Option 值的存在; Result 错误的存在
fn plus_one(x: Option<i32>) -> Option<i32> {
    match x {
        None => None,
        Some(i) => Some(i + 1),
    }
}

// Option 实现了Copy, 会move
let five = Some(5);
let six = plus_one(five);
let none = plus_one(None);

// unwrap解包Some()里包含的值
let six = six.unwrap();
}

解构数组
#![allow(unused)]
fn main() {
// 定长数组
let arr: [u16; 2] = [114, 514];
let [x, y] = arr;

assert_eq!(x, 114);
assert_eq!(y, 514);
}
#![allow(unused)]
fn main() {
// 不定长数组
let arr: &[u16] = &[114, 514];

if let [x, ..] = arr {
    assert_eq!(x, &114);
}

if let &[.., y] = arr {
    assert_eq!(y, 514);
}

let arr: &[u16] = &[];

assert!(matches!(arr, [..]));   // [..] 就是个空数组
assert!(!matches!(arr, [x, ..]));   // x is not in scope
}

5.4.1 @前绑定后解构(Rust 1.56 新增)

#[derive(Debug)]
struct Point {
    x: i32,
    y: i32,
}

fn main() {
    // 绑定新变量 `p`，同时对 `Point` 进行解构
    let p @ Point {x: px, y: py } = Point {x: 10, y: 23};
    println!("x: {}, y: {}", px, py);
    println!("{:?}", p);


    let point = Point {x: 10, y: 5};
    if let p @ Point {x: 10, y} = point {
        println!("x is 10 and y is {} in {:?}", y, p);
    } else {
        println!("x was not 10 :(");
    }
}
